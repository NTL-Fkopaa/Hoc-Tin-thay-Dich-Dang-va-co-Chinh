#include <bits/stdc++.h>
#include <string>
using namespace std;
const int N =1e7+1;
int n,p[N];
void h(){
  for (int i=1;i<N;i++) p[i]=true;
  p[1]=false;
  for (int i =2;i*i<=N;i++){
    if(p[i]){
      for (int j=i*i;j<N;j+=i)p[j]=false;
    }
  }
}

int R(int x){
  string s=to_string(x);
  reverse(s.begin(),s.end());
  return  stoi(s);
}


int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    h();
    cin>>n;
    int k=0;  
    for(int i=2;i<=n;i++){
      int x=R(i);
      if(p[x]) 
        k++;
      }
    
    cout<<k;
    return 0;
}
