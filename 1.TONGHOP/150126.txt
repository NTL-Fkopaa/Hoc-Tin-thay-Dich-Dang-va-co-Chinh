--Đánh số trang--

#include <iostream>
using namespace std;

int scs(int n) {
    int res = 0;
    if (n == 0) return 1; 
    while (n > 0) {
        res++;
        n /= 10;
    }
    return res;
}

int main() {
    long long n; 
    cin>>n;

    long long m = 0;  
    int res = 0;

    while (m < n) {
        res++;
        m += scs(res);
    }

    if (m == n) {
        cout << res << endl;
    } else {
        cout << 0 << endl;
    }

    return 0;
}

--Gtri khác nhau--
#include <bits/stdc++.h>

using namespace std;

int main() {
    int n;
    cin >> n;

    int a[1000]; 
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    sort(a, a + n);

    for (int i = n - 1; i >= 0; i--) {
        if (i == n - 1 || a[i] != a[i + 1]) {
            cout << a[i] << " ";
        }
    }

    return 0;
}

--Xếp thùng sơn--
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m, k;
    char c1, c2;

    cin >> n >> c1;
    cin >> m >> c2;
    cin >> k;

    int result;

    if (c1 == 'R') {
        result = min(k, n);
    } else {
        result = max(0, k - n);
    }

    cout << result << endl;
}

--Kiện hàng--
#include <iostream>
using namespace std;
int main() {

    int n, a[10001];
    int sum = 0, moved = 0;

    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> a[i];
        sum += a[i];
    }

    int target = sum / n;

    for (int i = 0; i < n; i++) {
        if (a[i] > target) {
            moved += (a[i] - target);
        }
    }

    cout << moved;

    return 0;
}
