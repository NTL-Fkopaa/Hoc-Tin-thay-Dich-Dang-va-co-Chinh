using namespace std;
int a[1000007];int n;
int main() {
    cin>>n;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    int low = 0, h = n - 1;
    int r = n; 

    while (low <= h) {
        int mid = low + (h - low) / 2;

        if (a[mid] == 0) {
            r = mid;
            h = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << r;
    return 0;
}
