#include <iostream>
#include <algorithm> 
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n;
    cin>>n;
    int r = 0;
    for (int i = 0; i < n; i++) {
        int a;
        cin >> a;
        if (a > r) {
            r = a;
        }
        cout << r << (i == n - 1 ? "" : " ");
    }

    return 0;
}
