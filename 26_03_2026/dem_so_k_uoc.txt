#include <iostream>
using namespace std;

int h[100005]; 

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, k;
    if (!(cin >> n >> k)) return 0;

    for (int i = 1; i <= n; ++i) {
        for (int j = i; j <= n; j += i) {
            h[j]++;
        }
    }

    int result = 0;
    for (int i = 1; i <= n; ++i) {
        if (h[i] == k) {
            result++;
        }
    }

    cout << result << endl;

    return 0;
}
